<script>
// Admin Password
let password = "1234";

// Product List
let products = JSON.parse(localStorage.getItem("products")) || [];

// Login
function login() {
  if(document.getElementById("pass").value===password){
    document.getElementById("adminPanel").style.display="block";
    document.getElementById("loginBox").style.display="none";
    showAdmin();
  } else { alert("Wrong Password!"); }
}

// Save products
function save(){ localStorage.setItem("products", JSON.stringify(products)); }

// Add Product
function addProduct(){
  let p = {
    name: document.getElementById("name").value,
    price: document.getElementById("price").value,
    img: document.getElementById("img").value,
    category: document.getElementById("cat").value
  };
  products.push(p);
  save();
  showAdmin();
  showShop('all');
}

// Delete Product
function deleteProduct(i){
  products.splice(i,1);
  save();
  showAdmin();
  showShop('all');
}

// Show Admin Panel Products
function showAdmin(){
  let list=document.getElementById("adminList"); list.innerHTML="";
  products.forEach((p,i)=>{
    list.innerHTML+=`
    <div class="card">
      <img src="${p.img}">
      <h4>${p.name}</h4>
      <p>৳${p.price}</p>
      <p>${p.category}</p>
      <button onclick="deleteProduct(${i})">Delete</button>
    </div>`;
  });
}

// Show Shop Products for Customer
function showShop(cat){
  let list=document.getElementById("shopList"); list.innerHTML="";
  products.forEach((p)=>{
    if(cat==='all' || p.category===cat){
      list.innerHTML+=`
      <div class="card">
        <img src="${p.img}">
        <h4>${p.name}</h4>
        <p>৳${p.price}</p>
        <button onclick="order('${p.name}')">Order via WhatsApp</button>
      </div>`;
    }
  });
}

// Filter function
function filter(cat){ showShop(cat); }

// WhatsApp Order
function order(name){
  let text = "I want to order: " + name;
  window.open("https://wa.me/8801824518901 text=" + text);
}

// Initial display
showShop('all');
</script>
